  ```python
  # generate_exit_signal()内で利益率を計算
  ```

```python
def analyze_transactions(csv_path: str) -> dict:
    """all_transactions.csvから統計を計算"""
    import pandas as pd
    
    df = pd.read_csv(csv_path)
    
    total_trades = len(df)
    total_pnl = df['pnl'].sum()
    avg_return = df['return_pct'].mean() * 100
    wins = len(df[df['pnl'] > 0])
    win_rate = (wins / total_trades) * 100 if total_trades > 0 else 0
    max_profit = df['pnl'].max()
    max_loss = df['pnl'].min()
    
    return {
        'total_trades': total_trades,
        'total_pnl': total_pnl,
        'avg_return': avg_return,
        'win_rate': win_rate,
        'max_profit': max_profit,
        'max_loss': max_loss
    }
```

#### 4. 最適パラメータ特定機能
```python
def find_best_params(results: list) -> dict:
    """
    総損益+勝率の複合評価で最適パラメータを特定
    
    評価式: score = total_pnl + (win_rate * 1000)
    理由: 総損益重視だが、勝率50%と60%の差（+10,000円相当）も考慮
    """
    best_score = -float('inf')
    best_pattern = None
    
    for result in results:
        score = result['total_pnl'] + (result['win_rate'] * 1000)
        if score > best_score:
            best_score = score
            best_pattern = result
    
    return best_pattern
```
